**步骤 1：新建 modelcraft-backend/internal/middleware/chi_scope_auth.go**

```go
package middleware

import (
    "encoding/json"
    "net/http"
)

const headerTokenScope = "X-Token-Scope"

// RequireScope 返回一个 chi 中间件，校验请求的 X-Token-Scope header 是否在 allowedScopes 中。
// 若 header 为空（SkipValidation 模式或 X-Internal-Token 模式），直接放行。
// 若 header 不在允许列表，返回 403 JSON 错误。
func RequireScope(allowedScopes ...string) func(http.Handler) http.Handler {
    allowed := make(map[string]bool, len(allowedScopes))
    for _, s := range allowedScopes {
        allowed[s] = true
    }
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            scope := r.Header.Get(headerTokenScope)
            if scope == "" {
                // header 为空：SkipValidation / InternalToken 模式，放行
                next.ServeHTTP(w, r)
                return
            }
            if !allowed[scope] {
                writeScopeError(w, http.StatusForbidden, "INSUFFICIENT_SCOPE", "token scope is not allowed for this endpoint")
                return
            }
            next.ServeHTTP(w, r)
        })
    }
}

func writeScopeError(w http.ResponseWriter, status int, code, message string) {
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(status)
    _ = json.NewEncoder(w).Encode(map[string]interface{}{
        "error": map[string]string{
            "code":    code,
            "message": message,
        },
    })
}
```

**步骤 2：在 modelcraft-backend/internal/app/auth/token_service.go 末尾新增 ExchangeToken 方法**

```go
// ExchangeToken 用有效的 Org Token 换取 Project Token。
// 不做 RBAC 校验；具体 project 可访问范围由下游 RBAC 决定。
// orgToken 必须是 scope="org" 的有效 JWT，否则返回业务错误。
func (s *TokenService) ExchangeToken(ctx context.Context, orgToken string) (string, time.Time, error) {
    claims, err := s.jwtSigner.ParsePlatformClaims(orgToken)
    if err != nil {
        return "", time.Time{}, bizerrors.NewErrorFromContext(ctx, bizerrors.AuthUnauthorized, "invalid org token: "+err.Error())
    }
    if claims.Scope != domainauth.TokenScopeOrg {
        return "", time.Time{}, bizerrors.NewErrorFromContext(ctx, bizerrors.AuthParamInvalid, "token scope must be 'org' for exchange, got: "+claims.Scope)
    }
    projectToken, err := s.jwtSigner.IssueAccessToken(claims.UserID, claims.OrgName, domainauth.TokenScopeProject)
    if err != nil {
        return "", time.Time{}, bizerrors.WrapError(err, bizerrors.SystemError, "exchange: issue project token")
    }
    expiresAt := time.Now().Add(time.Duration(s.jwtSigner.TTLSeconds()) * time.Second)
    return projectToken, expiresAt, nil
}
```

**重要**：`domainauth` 已是 token_service.go 中的 import 别名（参见文件中 `domainauth.TokenScopeOrg` 的现有用法）。

```go
// HandleExchange 处理 POST /api/auth/exchange — 凭 Org Token 换取 Project Token。
// Authorization 头中的 Bearer token 必须是有效的 scope=org JWT。
func (h *Handler) HandleExchange(w http.ResponseWriter, r *http.Request) {
    requestID := ctxutils.GetRequestID(r.Context())

    // 从 Authorization header 提取 Bearer token
    authHeader := r.Header.Get("Authorization")
    const bearerPrefix = "Bearer "
    if len(authHeader) <= len(bearerPrefix) || authHeader[:len(bearerPrefix)] != bearerPrefix {
        writeAuthError(w, http.StatusUnauthorized, requestID, "UNAUTHORIZED", "Authorization Bearer token required")
        return
    }
    orgToken := authHeader[len(bearerPrefix):]

    projectToken, expiresAt, err := h.tokenService.ExchangeToken(r.Context(), orgToken)
    if err != nil {
        h.handleBusinessError(w, r, requestID, err, "Exchange token failed")
        return
    }

    writeJSON(w, http.StatusOK, map[string]any{
        "requestId":   requestID,
        "accessToken": projectToken,
        "expiresAt":   expiresAt.UTC().Format("2006-01-02T15:04:05Z07:00"),
        "expiresIn":   h.tokenService.AccessTokenTTLSeconds(),
    })
}
```

**注意**：`AccessTokenTTLSeconds()` 如不存在，可使用 `h.tokenService` 的其他方式或直接不返回该字段（只返回 accessToken + expiresAt + requestId 即可满足要求）。检查 token_service.go 是否有 TTL 相关方法，如无则去掉 expiresIn 字段。

**步骤 4：gateway auth/handler.go — 新增 ExchangeToken 方法**

在 handler.go 末尾（EndUserMe 之后）追加：

```go
// ExchangeToken 透传 POST /api/auth/exchange 到 backend。
// 直接透传客户端的 Authorization Bearer token（backend 自行验证）。
// 不做 JWT 校验，不管理 cookie。
func (h *Handler) ExchangeToken(w http.ResponseWriter, r *http.Request) {
    authHeader := r.Header.Get("Authorization")
    if authHeader == "" {
        writeError(w, http.StatusUnauthorized, "MISSING_TOKEN", "Authorization header required")
        return
    }

    raw, err := h.postBackendRaw(r.Context(), "/api/auth/exchange", r.Body, func(req *http.Request) {
        req.Header.Set("Authorization", authHeader)
    })
    if err != nil {
        proxyBackendError(w, err)
        return
    }

    writeJSON(w, http.StatusOK, json.RawMessage(raw))
}
```

**注意**：当前 `postBackendRaw(ctx, path, body)` 只有三个参数（不支持 setHeaders 回调）。检查现有实现：
- 如果 `postBackendRaw` 不支持 header 注入，需要使用 `postBackendRawWithHeaders` 或参考 `getBackendRaw` 的 `setHeaders func(*http.Request)` 模式，为 `postBackendRaw` 新增重载或直接内联实现。
- 最简做法：参考 `EndUserMe` 中的 `getBackendRaw(ctx, path, setHeaders)` 模式，新增一个私有方法 `postBackendRawWithHeaders` 或直接内联 POST 逻辑。

推荐内联实现（避免修改现有方法签名）：

```go
func (h *Handler) ExchangeToken(w http.ResponseWriter, r *http.Request) {
    authHeader := r.Header.Get("Authorization")
    if authHeader == "" {
        writeError(w, http.StatusUnauthorized, "MISSING_TOKEN", "Authorization header required")
        return
    }

    req, err := http.NewRequestWithContext(r.Context(), http.MethodPost, h.backendURL+"/api/auth/exchange", r.Body)
    if err != nil {
        writeError(w, http.StatusInternalServerError, "INTERNAL_ERROR", "failed to create request")
        return
    }
    req.Header.Set("Content-Type", "application/json")
    req.Header.Set("Authorization", authHeader)
    if reqID := chiMiddleware.GetReqID(r.Context()); reqID != "" {
        req.Header.Set("X-Request-Id", reqID)
    }

    resp, err := h.httpClient.Do(req)
    if err != nil {
        writeError(w, http.StatusBadGateway, "UPSTREAM_ERROR", err.Error())
        return
    }
    defer resp.Body.Close()

    raw, _ := io.ReadAll(resp.Body)
    if resp.StatusCode >= 400 {
        w.Header().Set("Content-Type", "application/json")
        w.WriteHeader(resp.StatusCode)
        _, _ = w.Write(raw)
        return
    }
    writeJSON(w, http.StatusOK, json.RawMessage(raw))
}
```
