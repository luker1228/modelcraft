```go
// ParsePlatformClaims 使用本 signer 的 ES256 公钥验证并解析 PlatformClaims JWT。
// 验证内容：ES256 签名、token 有效期、以及 claims.Validate()（issuer + scope 合法性）。
// 使用场景：Exchange 端点验证 Org Token；scope 中间件（可选）二次验证。
func (s *JWTSigner) ParsePlatformClaims(tokenStr string) (*PlatformClaims, error) {
    token, err := jwt.ParseWithClaims(tokenStr, &PlatformClaims{}, func(t *jwt.Token) (interface{}, error) {
        if _, ok := t.Method.(*jwt.SigningMethodECDSA); !ok {
            return nil, fmt.Errorf("unexpected signing method: %v", t.Header["alg"])
        }
        return &s.privateKey.PublicKey, nil
    })
    if err != nil || !token.Valid {
        return nil, fmt.Errorf("jwt_signer: parse claims: %w", err)
    }
    claims, ok := token.Claims.(*PlatformClaims)
    if !ok {
        return nil, errors.New("jwt_signer: unexpected claims type")
    }
    if err := claims.Validate(); err != nil {
        return nil, fmt.Errorf("jwt_signer: invalid claims: %w", err)
    }
    return claims, nil
}
```

func TestJWTSigner_ParsePlatformClaims(t *testing.T) {
    signer, err := GenerateDevSigner()
    require.NoError(t, err)

    t.Run("valid org token", func(t *testing.T) {
        token, err := signer.IssueAccessToken("user-1", "acme", TokenScopeOrg)
        require.NoError(t, err)
        claims, err := signer.ParsePlatformClaims(token)
        require.NoError(t, err)
        assert.Equal(t, "user-1", claims.UserID)
        assert.Equal(t, "acme", claims.OrgName)
        assert.Equal(t, TokenScopeOrg, claims.Scope)
    })

    t.Run("valid project token", func(t *testing.T) {
        token, err := signer.IssueAccessToken("user-2", "beta", TokenScopeProject)
        require.NoError(t, err)
        claims, err := signer.ParsePlatformClaims(token)
        require.NoError(t, err)
        assert.Equal(t, TokenScopeProject, claims.Scope)
    })

    t.Run("empty token", func(t *testing.T) {
        _, err := signer.ParsePlatformClaims("")
        require.Error(t, err)
    })

    t.Run("tampered token", func(t *testing.T) {
        token, err := signer.IssueAccessToken("user-1", "acme", TokenScopeOrg)
        require.NoError(t, err)
        // 修改 token 最后几个字符模拟篡改

        require.Error(t, err)
    })

    t.Run("token from different signer", func(t *testing.T) {
        otherSigner, _ := GenerateDevSigner()
        token, err := otherSigner.IssueAccessToken("user-1", "acme", TokenScopeOrg)
        require.NoError(t, err)
        _, err = signer.ParsePlatformClaims(token)
        require.Error(t, err)
    })
}

<threat_model>
## 信任边界

| 边界 | 描述 |
|------|------|
| 外部请求 → Gateway | 外部可伪造任意 HTTP Header（含 X-Token-Scope） |
| Gateway → Backend | 内网转发，但 X-Token-Scope 须 Gateway 覆写以防伪造 |

## STRIDE 威胁登记

| 威胁 ID | 类别 | 组件 | 处置 | 缓解方案 |
|---------|------|------|------|---------|
| T-02-01 | 欺骗（Spoofing） | proxy/handler.go director | 缓解 | 先 `req.Header.Del("X-Token-Scope")` 再 `Set`，外部携带的同名 header 会被覆盖 |
| T-02-02 | 篡改（Tampering） | jwt_signer.ParsePlatformClaims | 缓解 | ES256 签名验证；使用 `&s.privateKey.PublicKey` 直接验签，无 PEM 中间环节 |
| T-02-03 | 信息泄露（Info Disclosure） | Claims.Scope | 接受 | scope 不是秘密（JWT payload 可 base64 解码查看），无 PII 风险 |
| T-02-04 | 拒绝服务（DoS） | ParsePlatformClaims | 接受 | 由 Gateway JWT 验证提前拦截无效 token；ParsePlatformClaims 仅用于 exchange 端点（Wave 2），非热路径 |
</threat_model>
