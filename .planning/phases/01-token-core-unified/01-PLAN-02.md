// 使用与平台管理员相同的 JWTSigner，签发 PlatformClaims（scope=org）。
type endUserJWTIssuer struct {

func (i *endUserJWTIssuer) IssueEndUserToken(
    _ context.Context,
    input appEnduser.EndUserTokenIssueInput,
) (*appEnduser.EndUserTokenIssueResult, error) {
    if i.signer == nil {
        return nil, fmt.Errorf("end-user jwt signer is nil")

    }
    return &appEnduser.EndUserTokenIssueResult{
        AccessToken: accessToken,
        ExpiresAt:   now.Add(time.Duration(i.signer.TTLSeconds()) * time.Second),
    }, nil
}

```go
// parseEndUserJWT 使用 ES256 公钥验证端用户 Bearer token，解析为 PlatformClaims。
func (h *AuthHandler) parseEndUserJWT(tokenStr string) (*domainAuth.PlatformClaims, error) {
    pubKeyPEM, err := h.jwtSigner.PublicKeyPEM()
    if err != nil {
        return nil, fmt.Errorf("get public key: %w", err)
    }
    block, _ := pem.Decode([]byte(pubKeyPEM))
    if block == nil {
        return nil, fmt.Errorf("failed to decode public key PEM")
    }
    pub, err := x509.ParsePKIXPublicKey(block.Bytes)
    if err != nil {
        return nil, fmt.Errorf("parse public key: %w", err)
    }
    token, err := jwt.ParseWithClaims(tokenStr, &domainAuth.PlatformClaims{}, func(t *jwt.Token) (interface{}, error) {
        if _, ok := t.Method.(*jwt.SigningMethodECDSA); !ok {
            return nil, fmt.Errorf("unexpected signing method: %v", t.Header["alg"])
        }
        return pub, nil
    })
    if err != nil || !token.Valid {
        return nil, fmt.Errorf("invalid token")
    }
    claims, ok := token.Claims.(*domainAuth.PlatformClaims)
    if !ok || claims.OrgName == "" || claims.Subject == "" {
        return nil, fmt.Errorf("missing required claims")
    }
    return claims, nil
}
```

## 信任边界

| 边界 | 描述 |
|------|------|
| HTTP Request → RuntimeAuthMiddleware | Bearer token 来自客户端，需验证签名 + issuer |
| endUserJWTIssuer → JWTSigner | 接口层调用领域层签发，私钥仅在 backend 持有 |
| auth_handler.go /me → parseEndUserJWT | Bearer token 从 Authorization header 取，ES256 公钥验证 |

## STRIDE 威胁登记

| 威胁 ID | 类别 | 组件 | 处置 | 缓解措施 |
|---------|------|------|------|----------|
| T-02-01 | 欺骗（S） | endUserJWTIssuer HMAC→ES256 迁移 | 缓解 | 旧 HMAC secret 仅 gateway 和 backend 双方持有，迁移后 secret 不再使用；ES256 私钥只在 backend |
| T-02-02 | 篡改（T） | Refresh 路径 orgName 从 membership 查询 | 接受 | membership 是受控的 DB 数据；查询失败时 orgName 为空，IssueAccessToken 会返回 error 阻断签发 |
| T-02-03 | 拒绝服务（D） | RuntimeAuthMiddleware issuer 硬切 | 接受 | 硬切导致旧 mc-enduser token 立即失效（401），设计决策，TTL 1h 内自然过期 |
| T-02-04 | 信息泄露（I） | auth_handler.go parseEndUserJWT 用 PublicKeyPEM | 缓解 | 公钥是公开数据，不存在泄露风险；私钥不在 auth_handler 中 |
